async function handleLogin() { const birthday = document.getElementById('birthday').value; if (!birthday || birthday.length < 8) return alert("請輸入正確的 8 位數生日"); const btn = document.getElementById('loginBtn'); btn.innerText = "驗證中..."; btn.disabled = true; try { // 使用您的 API 連結，並指定搜尋 Birthday 欄位 const response = await fetch(`https://sheetdb.io/api/v1/poo9epu17583p/search?Birthday=${birthday}`); const data = await response.json(); if (data.length > 0) { // 登入成功，儲存整筆客人資訊 (包含 ID, NAME, Status) localStorage.setItem('user', JSON.stringify(data[0])); // 跳轉至主頁面 window.location.href = 'src/main.html'; } else { alert("查無此身分資訊，請確認生日格式是否為 YYYYMMDD（如 19990922）。"); } } catch (e) { alert("網路連線異常，請稍後再試。"); } finally { btn.innerText = "進入系統"; btn.disabled = false; } }
